async function startConversion() { if (!uploadedFile) { showError("Please upload an image first."); return; } if (uploadedFile.size > 10 * 1024 * 1024) { showError("File is too large. Max allowed size is 10MB."); return; } // prevent duplicate submissions if (convertBtn) convertBtn.disabled = true; if (cancelBtn) cancelBtn.disabled = false; // reset state for new job if (pollTimer) { clearInterval(pollTimer); pollTimer = null; } currentJobId = null; socketReadyTimedOut = false; const socket = window.__socket; const connected = !!(socket && socket.connected && window.__socketId); // allow socket to connect briefly, then fallback to polling if (!connected) { await new Promise((resolve) => { let done = false; const finish = () => { if (done) return; done = true; resolve(); }; const t = setTimeout(() => { socketReadyTimedOut = true; finish(); }, 3000); if (socket) { const onConnect = () => { clearTimeout(t); socketReadyTimedOut = false; socket.off("connect", onConnect); finish(); }; socket.on("connect", onConnect); } else { finish(); } }); } try { setStatus("Submitting conversion job..."); const fd = buildFormData("convert"); if (window.__socketId) fd.set("socketId", window.__socketId); const res = await fetch("/generate-embroidery", { method: "POST", body: fd }); const data = await res.json().catch(() => ({})); await handleAuthOrThrow(res, data); const jobId = data.jobId || data.id; if (!jobId) throw new Error("Server did not return a jobId."); currentJobId = jobId; // subscribe only after currentJobId is assigned if (window.__socket && window.__socket.connected) { window.__socket.emit("subscribe", currentJobId); } else { socketReadyTimedOut = true; } // polling fallback (or primary if socket unavailable) if (socketReadyTimedOut) { if (pollTimer) clearInterval(pollTimer); pollTimer = setInterval(() => { if (!currentJobId) return; pollJobStatus(currentJobId); }, 1000); pollJobStatus(currentJobId); } setStatus("Job queued..."); } catch (err) { showError(err.message || "Conversion failed."); if (convertBtn) convertBtn.disabled = false; if (cancelBtn) cancelBtn.disabled = true; } }
